SELECT _OPEN.SUBSCRIBERKEY AS AC_PERSONCONTACTID
FROM _CLICK
INNER JOIN _SENT ON _SENT.JOBID = _OPEN.JOBID
    AND _OPEN.LISTID = _SENT.LISTID
    AND _OPEN.BATCHID = _SENT.BATCHID
    AND _OPEN.SUBSCRIBERKEY = _SENT.SUBSCRIBERKEY
    AND _OPEN.ISUNIQUE = '1'
INNER JOIN _JOURNEYACTIVITY ON _JOURNEYACTIVITY.JOURNEYACTIVITYOBJECTID = _SENT.TRIGGERERSENDDEFINITIONOBJECTID
    AND _JOURNEYACTIVITY.ACTIVITYTYPE IN ('EMAIL','EMAILV2')
INNER JOIN _JOURNEY ON _JOURNEY.VERSIONID = _JOURNEYACTIVITY.VERSIONID
    AND _JOURNEY.JOURNEYNAME IN (##journey_name##)
